13// XOR: If the corresponding bits in the two inputs are different, the result is 1, otherwise, it's 0.
14// meaning applying XOR operation with the same key again decrypts the message.
15char* encrypt(const char* message, const char* key) {
16 size_t messageLength = strlen(message);
17 size_t keyLength = strlen(key);
18 char* encryptedMessage = (char*)malloc(messageLength + 1);
19
20 // Iterate over each character in the message
21 for (size_t i = 0; i < messageLength; i++) {
22 // Perform XOR operation between the current character of the message and the corresponding character of the key.
23 // The modulo operation ensures that if the key is shorter than the message, it will be repeated cyclically
24 encryptedMessage[i] = message[i] ^ key[i % keyLength];
25 }
26 // Add null terminator
27 encryptedMessage[messageLength] = '\0';
28
29 return encryptedMessage;
30}
